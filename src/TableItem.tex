\newcommand{\rReference}[6]{
    \begin{flushleft}
        \vspace{-5pt}
        \begin{tabularx}{\textwidth}{@{}X X@{\hspace{8pt}}}
            \begin{tabular}[t]{@{}l@{}}
                \hspace{8pt}\textbf{#1} \\ % name
                \hspace{8pt}{\textit{#2}} \\ % job title
                \hspace{8pt}{#3} \\ % company
                % skips phone number if empty
                \if\relax#4\relax
                    \hspace{8pt}{\href{mailto:#5}{#5}} \\ % email
                \else
                    \hspace{8pt}{\href{tel:#4}{#4}} \\ % phone
                    \hspace{8pt}{\href{mailto:#5}{#5}} \\ % email
                \fi
            \end{tabular} & {#6}
        \end{tabularx}
        \vspace{-5pt}

        \hrulefill
    \end{flushleft}
}
